git commands:

git init
# Make folder a git
git clone <url>

git log
git log --graph
# Show log with branches

git add <filename>
git rm <filename>
git status

git diff <old> <new>
# Compare commits
git diff
# Compare working directory with stage
git diff --staged
# Compare stage with current commit
git show <id>
# Compare changes of this commit from its parent

git commit
# https://udacity.github.io/git-styleguide/
git branch
# Show branches
git branch <name>
# Create a branch label
git branch -d <name>
# Delete a branch label

git checkout <label>
# Go checkout this branch label
git merge <branch1> <branch2>

git remote
git remote add <origin> <url>
# origin is a default name for 1 remote
git remove -v
git push
